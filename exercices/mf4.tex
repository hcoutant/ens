# These definitions are overriden by the top-level makefile
# You shouldn't have to modify this file.

LATEX   = latex
DVIPS   = dvips

# Our default rules
.SUFFIXES:      .tex .dvi .ps

.tex.dvi:
        $(LATEX) $*

.dvi.ps:
        $(DVIPS) -f $*.dvi > $*.ps

all:    install.ps manuel.ps interne.ps

clean:
        rm -f *.ps *.aux *.dvi *.log *.idx texput.log

install.ps: install.dvi
install.dvi: install.tex

manuel.ps: manuel.dvi
manuel.dvi: manuel.tex ch1.tex ch2.tex ch3.tex

interne.ps: interne.dvi
interne.dvi: interne.tex
